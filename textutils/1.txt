
def getSolution(H, P):
    ans = 0
    P = P * 2
    if P > H:
        ans = 1
    return ans

try:
    if __name__ == "__main__":
        T = int(input())
        for _ in range(T):
            H, P = input().split()
            H, P = int(H), int(P)
            ans = getSolution(H, P)
            print(ans)
except EOFError as e:
    pass
